        let cart = [];

        function renderProducts(productsToShow = products) {
            const productsEl = document.getElementById('products');
            productsEl.innerHTML = productsToShow.map(product => `
                <div class="product" data-character="${product.character}" data-length="${product.length}">
                    <img src="${product.img}" alt="${product.name}" loading="lazy">
                    <div class="product-info">
                        <h3>${product.name}</h3>
                        <p>Length: ${product.length}</p>
                        <p class="price">₹${product.price}</p>
                        <button onclick="addToCart(${product.id})">Add to Cart</button>
                    </div>
                </div>
            `).join('');
        }

        function filterProducts() {
            const characterFilter = document.getElementById('characterFilter').value;
            const lengthFilter = document.getElementById('lengthFilter').value;
            let filtered = products;
            if (characterFilter) filtered = filtered.filter(p => p.character === characterFilter);
            if (lengthFilter) filtered = filtered.filter(p => p.length === lengthFilter);
            renderProducts(filtered);
        }

        function addToCart(id) {
            const product = products.find(p => p.id === id);
            const cartItem = cart.find(item => item.id === id);
            if (cartItem) {
                cartItem.quantity += 1;
            } else {
                cart.push({ ...product, quantity: 1 });
            }
            updateCartUI();
            alert(`${product.name} added to cart!`);
        }

        function updateCartUI() {
            document.getElementById('cartCount').textContent = cart.reduce((sum, item) => sum + item.quantity, 0);
        }

        function renderCart() {
            const cartItemsEl = document.getElementById('cartItems');
            cartItemsEl.innerHTML = cart.map(item => `
                <div class="cart-item">
                    <span>${item.name} (${item.length}) x${item.quantity}</span>
                    <span>₹${item.price * item.quantity}</span>
                </div>
            `).join('');
            document.getElementById('cartTotal').textContent = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
        }

        function checkout() {
            if (cart.length === 0) return alert('Cart is empty!');
            alert(`Order Summary: ${cart.reduce((sum, item) => sum + (item.price * item.quantity), 0)} ₹\nContact us at paperkatanas@email.com to complete your DIY paper katana order!`);
            cart = [];
            updateCartUI();
            document.getElementById('cart').style.display = 'none';
        }

        // Event Listeners
        document.getElementById('characterFilter').addEventListener('change', filterProducts);
        document.getElementById('lengthFilter').addEventListener('change', filterProducts);
        document.getElementById('cartBtn').addEventListener('click', () => {
            document.getElementById('cart').style.display = 'block';
            renderCart();
        });
        document.getElementById('closeCart').addEventListener('click', () => {
            document.getElementById('cart').style.display = 'none';
        });

        // Initial render
        renderProducts();
